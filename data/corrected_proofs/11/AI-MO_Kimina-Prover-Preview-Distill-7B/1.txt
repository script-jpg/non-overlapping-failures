-- We have $2x = 5y$, so $y = \frac{2}{5}x$.
have h₃ : y = 2 / 5 * x := by linarith

-- We have $7y = 10z$, so $z = \frac{7}{10}y$.
have h₄ : z = 7 / 10 * y := by linarith

-- Substituting $y = \frac{2}{5}x$ into $z = \frac{7}{10}y$, we get $z = \frac{7}{10} \cdot \frac{2}{5}x = \frac{7}{25}x$.
have h₅ : z = 7 / 25 * x := by
  rw [h₄, h₃]
  ring

-- Therefore, $\frac{z}{x} = \frac{7}{25}$.
have h₆ : z / x = 7 / 25 := by
  have h₇ : x ≠ 0 := h₀
  field_simp [h₇]
  rw [h₅]
  ring

exact h₆
