have h : ∀ n : ℕ, 11 ∣ 10 ^ n - (-1 : ℤ) ^ n := by
  intro n
  induction' n with n ih
  · norm_num -- Base case: when n = 0, 10^0 - (-1)^0 = 1 - 1 = 0, which is divisible by 11.
  · cases' n with n
    · norm_num -- Base case: when n = 1, 10^1 - (-1)^1 = 10 - (-1) = 11, which is divisible by 11.
    · simp [pow_add, pow_mul, Int.emod_eq_of_lt] at ih ⊢
      -- Inductive step: use the fact that 10^n - (-1)^n is divisible by 11 to show 10^(n+1) - (-1)^(n+1) is also divisible by 11.
      omega
apply h
