-- We find the number of integers between 15 and 85 that are divisible by 20.
have : (Finset.filter (fun x => 20 ∣ x) (Finset.Icc 15 85)) = {20, 40, 60, 80} := by
  -- We prove this by showing both inclusions.
  ext x
  constructor

  -- Firstly, an integer x in the set on the left is in the set on the right.
  · intro hx
    simp at hx ⊢
    obtain ⟨⟨hx1, hx2⟩, hx3⟩ := hx
    -- We know that 15 ≤ x ≤ 85 and 20 ∣ x.
    -- The integers between 15 and 85 that are divisible by 20 are 20, 40, 60, and 80.
    omega

  -- Secondly, an integer x in the set on the right is in the set on the left.
  · intro hx
    simp at hx ⊢
    obtain rfl | rfl | rfl | rfl := hx
    all_goals
      -- For each of the four integers 20, 40, 60, and 80, we show that it satisfies the two conditions.
      simp

-- We substitute the set on the left with the set on the right and simplify.
rw [this]
simp
