-- We need to find the count of integers between 15 and 85 (inclusive) that are divisible by 20.

-- The integers divisible by 20 between 15 and 85 are 20, 40, 60, and 80.
-- Thus, there are exactly 4 such integers.

-- The proof proceeds by showing that the filtered finset equals {20, 40, 60, 80} and then computing its cardinality.
have : Finset.filter (fun x => 20 ∣ x) (Finset.Icc 15 85) = {20, 40, 60, 80} := by
  -- Use extensionality to show the equality of finsets.
  ext x
  constructor
  · -- Assume x is in the filtered finset.
    simp [Finset.mem_filter, Finset.mem_Icc]
    intro hx hx2
    omega
  · -- Assume x is in {20, 40, 60, 80}.
    simp [Finset.mem_insert, Finset.mem_singleton]
    omega
rw [this]
-- The cardinality of the finset {20, 40, 60, 80} is 4.
simp
