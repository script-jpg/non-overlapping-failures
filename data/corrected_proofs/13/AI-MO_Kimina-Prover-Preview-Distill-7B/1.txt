-- Show that the product contains the factor 5 and the factor 2.
have h₁ : 1 * 3 * 5 * 7 * 9 * 11 * 13 % 2 = 1 := by
  norm_num
have h₂ : 1 * 3 * 5 * 7 * 9 * 11 * 13 % 5 = 0 := by
  norm_num

-- Show that the product is divisible by 5, and hence by 10.
have h₃ : 1 * 3 * 5 * 7 * 9 * 11 * 13 % 10 = 5 := by
  omega

-- Conclude that the ones digit is 5.
exact h₃
