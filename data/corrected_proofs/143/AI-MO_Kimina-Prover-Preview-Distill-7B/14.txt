-- Solve for $a$: $\dfrac{8^{-1}}{4^{-1}}-a^{-1}=1$. Show that it is -2.-/
-- Calculate $\dfrac{8^{-1}}{4^{-1}}$.
have h : (8 : ℝ)⁻¹ / 4⁻¹ = 1 / 2 := by
  -- Calculate $\dfrac{8^{-1}}{4^{-1}}$.
  calc
    (8 : ℝ)⁻¹ / 4⁻¹ = (4 / 8) := by
      field_simp
    _ = 1 / 2 := by
      norm_num

-- Calculate $1 / 2 - a^{-1} = 1$.
rw [h] at h₁
field_simp at h₁
-- Calculate $a = -2$.
linarith only [h₁]
