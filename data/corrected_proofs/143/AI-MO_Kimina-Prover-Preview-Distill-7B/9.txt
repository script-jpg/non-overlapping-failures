-- Solve for $a$: $\dfrac{8^{-1}}{4^{-1}}-a^{-1}=1$. Show that it is -2.-/
-- $\dfrac{8^{-1}}{4^{-1}} = \dfrac{1}{8} \div \dfrac{1}{4} = \dfrac{1}{8} \times \dfrac{4}{1} = \dfrac{4}{8} = \dfrac{1}{2}$
have h₂ : (8⁻¹ / 4⁻¹ : ℝ) = 1 / 2 := by
  norm_num
rw [h₂] at h₁
-- $1 / 2 - a^{-1} = 1 \implies -a^{-1} = 1 - 1 / 2 \implies -a^{-1} = 1 / 2 \implies a^{-1} = -1 / 2 \implies a = -2$
field_simp at h₁
linarith
