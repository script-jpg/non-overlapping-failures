-- Solve for $a$: $\dfrac{8^{-1}}{4^{-1}}-a^{-1}=1$. Show that it is -2.-/
-- $\dfrac{8^{-1}}{4^{-1}} = \dfrac{\frac{1}{8}}{\frac{1}{4}} = \dfrac{1}{8} \cdot \dfrac{4}{1} = \dfrac{4}{8} = \dfrac{1}{2}$
have h₂ : (8⁻¹ / 4⁻¹ : ℝ) = 1 / 2 := by
  norm_num
rw [h₂] at h₁
-- $1/2 - a^{-1} = 1 \Rightarrow a^{-1} = 1/2 - 1 = -1/2$
have h₃ : a⁻¹ = -1 / 2 := by linarith
-- $a^{-1} = -1/2 \Rightarrow a = -2$
have h₄ : a = -2 := by
  field_simp at h₃ ⊢
  linarith
exact h₄
