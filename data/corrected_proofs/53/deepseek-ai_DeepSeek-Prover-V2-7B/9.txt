have h₁ : Nat.gcd (21 * n + 4) (14 * n + 3) = 1 := by
  -- Use the fact that the GCD of two numbers is the same as the GCD of the smaller number and the remainder of the division of the larger number by the smaller number.
  rw [Nat.gcd_comm]
  have h₂ : 21 * n + 4 = 14 * n + 3 + (7 * n + 1) := by ring
  rw [h₂]
  have h₃ : 14 * n + 3 = 2 * (7 * n + 1) + 1 := by ring
  rw [h₃]
  -- Use the fact that the GCD of two numbers is the same as the GCD of the smaller number and the remainder of the division of the larger number by the smaller number.
  simp [Nat.gcd_comm, Nat.gcd_assoc, Nat.gcd_one_right]
  <;> omega
exact h₁
