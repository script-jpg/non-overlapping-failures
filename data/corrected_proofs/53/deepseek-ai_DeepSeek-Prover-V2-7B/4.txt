have h₁ : Nat.gcd (21 * n + 4) (14 * n + 3) = 1 := by
  rw [Nat.gcd_comm]
  -- Use the Euclidean algorithm to find the gcd of 21n + 4 and 14n + 3
  rw [Nat.gcd_comm]
  have h₂ : 21 * n + 4 = 14 * n + 3 + (7 * n + 1) := by ring
  rw [h₂]
  have h₃ : 14 * n + 3 = 2 * (7 * n + 1) + 1 := by ring
  rw [h₃]
  -- Apply the Euclidean algorithm
  norm_num
  <;> simp_all [Nat.gcd_comm, Nat.gcd_assoc, Nat.gcd_one_right]
  <;> omega
exact h₁
