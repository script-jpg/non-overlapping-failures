have h₂ : 0 ≤ 19 + 3 * y := by
  -- Since the square root of a non-negative number is non-negative, 19 + 3y must be non-negative.
  apply le_of_not_lt
  intro h
  rw [Real.sqrt_eq_zero_of_nonpos (by linarith)] at h₁
  norm_num at h₁
  <;> linarith

have h₃ : (Real.sqrt (19 + 3 * y)) ^ 2 = 7 ^ 2 := by
  rw [h₁]
  <;> norm_num

have h₄ : 19 + 3 * y = 49 := by
  rw [Real.sq_sqrt (by linarith)] at h₃
  <;> nlinarith

have h₅ : y = 10 := by
  linarith

exact h₅
