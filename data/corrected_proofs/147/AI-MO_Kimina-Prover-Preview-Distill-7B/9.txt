-- Let's compute $91^2$ using the formula $(a + b)^2 = a^2 + 2ab + b^2$ with $a = 90$ and $b = 1$.
calc 91 ^ 2 = (90 + 1) ^ 2 := by norm_num
         -- expanding $(90 + 1)^2$ using the binomial theorem
         _ = 90 ^ 2 + 2 * 90 * 1 + 1 ^ 2 := by ring
         -- computing $90^2 = 8100$, $2 \cdot 90 \cdot 1 = 180$, and $1^2 = 1$
         _ = 8100 + 180 + 1 := by norm_num
         -- adding these numbers to get $8281$
         _ = 8281 := by norm_num
