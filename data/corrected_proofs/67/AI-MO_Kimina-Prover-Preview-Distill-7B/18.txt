-- By substitution, we get $t = 3(9-2t)+1$.

  have h₂ : t = 3 * (9 - 2 * t) + 1 := by rw [← h₀, h₁]

  -- Hence, $t = 27 - 6t + 1$, or $t = 28 - 6t$.

  rw [mul_sub] at h₂; ring_nf at h₂

  -- Then, $7t = 28$, so $t = 4$.

  have h₃ : t = 4 := by linarith

  -- We can substitute $t = 4$ back into $s = 9 - 2t$ to get $s = 9 - 2(4) = 1$.

  rw [h₃] at h₀; ring_nf at h₀

  -- Hence, the intersection is at the point $(1,4)$.

  exact ⟨by linarith, h₃⟩
