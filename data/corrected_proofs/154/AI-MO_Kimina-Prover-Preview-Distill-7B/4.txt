-- the minimum value of a quadratic funtion ax^2+bx+c occurs at x = -b/(2a)
-- Here a=1, b=-14, so the minimum occurs at x = 14/2 = 7.
-- We can rewrite the expression by completing the square.
have : x ^ 2 - 14 * x + 3 = (x - 7) ^ 2 + (3 - 7 ^ 2) := by ring
rw [this]
-- We know that (x-7)^2 ≥ 0 since it is a square of a real number.
-- So the minimum value is attained when (x-7)^2 = 0, which is when x=7.
have : (x - 7) ^ 2 ≥ 0 := by exact sq_nonneg (x - 7)
