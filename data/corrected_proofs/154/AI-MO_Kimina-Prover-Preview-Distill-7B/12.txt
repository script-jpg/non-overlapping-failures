-- the quadratic $x^2 - 14x + 3$ has unique minimum since the coefficient of $x^2$ is positive.
have h : ∀ x : ℝ, x ^ 2 - 14 * x + 3 = (x - 7) ^ 2 + (3 - 7 ^ 2) := by intro x; ring

-- Thereby the minimum value of $x^2 - 14x + 3$ is $7 ^ 2 - 14 * 7 + 3$
have h' : ∀ x : ℝ, x ^ 2 - 14 * x + 3 ≥ 7 ^ 2 - 14 * 7 + 3 := by
  intro x
  rw [h x]
  have : (x - 7) ^ 2 ≥ 0 := by exact sq_nonneg (x - 7)
  linarith

exact h' x
