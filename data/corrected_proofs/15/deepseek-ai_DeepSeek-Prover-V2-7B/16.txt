have h₁ : a * b + |a - b| ≤ 1 := by
  cases' le_total 0 (a - b) with h h <;>
  cases' le_total 0 (a + b) with h₂ h₂ <;>
  cases' le_total 0 (a * b) with h₃ h₃ <;>
  simp_all only [abs_of_nonneg, abs_of_nonpos, sub_nonneg, sub_nonpos] <;>
  nlinarith [sq_nonneg (a - b), sq_nonneg (a + b), sq_nonneg (a + b - 1), sq_nonneg (a + b + 1),
    sq_nonneg (a - b - 1), sq_nonneg (a - b + 1)]
exact h₁
