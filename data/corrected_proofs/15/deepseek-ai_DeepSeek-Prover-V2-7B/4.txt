have h₁ : a ^ 2 + b ^ 2 = 1 := h₀
have h₂ : a * b + abs (a - b) ≤ 1 := by
  cases' le_total 0 (a - b) with h h <;> simp_all [abs_of_nonneg, abs_of_nonpos, sub_nonneg, sub_nonpos]
  all_goals
    nlinarith [sq_nonneg (a + b), sq_nonneg (a - b), sq_nonneg (a - b + 1), sq_nonneg (a + b - 1),
      sq_nonneg (a + b + 1), sq_nonneg (a + b - 2), sq_nonneg (a - b + 2), sq_nonneg (a - b - 2)]
exact h₂
