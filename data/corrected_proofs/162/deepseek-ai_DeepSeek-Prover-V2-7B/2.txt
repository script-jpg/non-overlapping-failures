have h : ∀ n : ℕ, -1 < x → 1 + n * x ≤ (1 + x) ^ n := by
  intro n hx
  induction n with
  | zero =>
    norm_num
  | succ n ih =>
    simp_all [pow_succ]
    nlinarith [mul_self_nonneg x, mul_self_nonneg (x + 1),
      mul_self_nonneg (x - 1), mul_self_nonneg (x + n * x),
      mul_self_nonneg (x - n * x), mul_self_nonneg (x + n * x + 1),
      mul_self_nonneg (x - n * x - 1)]
exact h n h₀
