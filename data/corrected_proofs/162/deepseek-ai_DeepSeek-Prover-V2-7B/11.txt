have h : ∀ n : ℕ, -1 < x → 1 + ↑n * x ≤ (1 + x) ^ (n : ℕ) := by
  intro n hx
  induction' n with n hn
  -- Base case: n = 0
  norm_num
  -- Inductive step: assume the statement holds for n, prove for n + 1
  simp_all [pow_succ]
  nlinarith [sq_nonneg x, pow_pos (sub_pos.mpr hx) n, mul_self_nonneg (x + 1),
    mul_self_nonneg (x - 1)]
apply h n h₀
<;> assumption
