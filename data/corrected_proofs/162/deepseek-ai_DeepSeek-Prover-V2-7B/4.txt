have h : ∀ n : ℕ, -1 < x → 1 + ↑n * x ≤ (1 + x) ^ (n : ℕ) := by
  intro n hx
  induction' n with n ih
  · simp -- Base case: when n = 0, the inequality simplifies to 1 ≤ 1, which is true.
  cases n with
  | zero =>
    norm_num -- Base case: when n = 1, the inequality simplifies to 1 + x ≤ 1 + x, which is true.
  | succ n =>
    simp_all [pow_succ, mul_add, add_mul, mul_one, mul_comm, mul_left_comm]
    -- Inductive step: Assume the inequality holds for n, prove it for n + 1.
    nlinarith [sq_nonneg x, pow_two_nonneg x, pow_two_nonneg (x + 1),
      pow_two_nonneg (x - 1), mul_nonneg (sub_nonneg.mpr hx.le) (sq_nonneg x),
      mul_nonneg (sub_nonneg.mpr hx.le) (sq_nonneg (x + 1)),
      mul_nonneg (sub_nonneg.mpr hx.le) (sq_nonneg (x - 1))]
exact h n h₀
