-- Use the binomial theorem to expand (1 + x)^n
have h : (1 + x) ^ n ≥ 1 + ↑n * x := by
  induction' h₁ with n hn
  -- Base case: n = 1
  norm_num
  simp_all [pow_succ, mul_add, mul_one, mul_comm]
  -- Inductive step: assume the inequality holds for n, prove it for n + 1
  nlinarith [sq_nonneg x, mul_self_nonneg (x + 1), mul_self_nonneg (x - 1)]
-- Use the established inequality to conclude the proof
linarith
