have h : ∀ n : ℕ, 0 < n → (1 + x) ^ n ≥ 1 + n * x := by
  intro n hn
  induction' hn with n hn IH
  · norm_num
    <;> nlinarith
  · cases n with
    | zero =>
      norm_num
      <;> nlinarith
    | succ n =>
      simp_all [pow_succ, mul_add, mul_one, mul_comm, mul_left_comm]
      <;> nlinarith [sq_nonneg x, sq_nonneg (x + 1), sq_nonneg (x - 1)]
exact h n h₁
