-- We calculate each floor function value separately.
have h₁ : Int.floor (10 * n) = 3 := by
  -- Substituting $n = \frac{1}{3}$, we get $10 \cdot \frac{1}{3} = \frac{10}{3} = 3\frac{1}{3}$.
  -- The greatest integer less than or equal to $3\frac{1}{3}$ is 3.
  rw [h₀]
  norm_num

have h₂ : Int.floor (100 * n) = 33 := by
  -- Substituting $n = \frac{1}{3}$, we get $100 \cdot \frac{1}{3} = \frac{100}{3} = 33\frac{1}{3}$.
  -- The greatest integer less than or equal to $33\frac{1}{3}$ is 33.
  rw [h₀]
  norm_num

have h₃ : Int.floor (1000 * n) = 333 := by
  -- Substituting $n = \frac{1}{3}$, we get $1000 \cdot \frac{1}{3} = \frac{1000}{3} = 333\frac{1}{3}$.
  -- The greatest integer less than or equal to $333\frac{1}{3}$ is 333.
  rw [h₀]
  norm_num

have h₄ : Int.floor (10000 * n) = 3333 := by
  -- Substituting $n = \frac{1}{3}$, we get $10000 \cdot \frac{1}{3} = \frac{10000}{3} = 3333\frac{1}{3}$.
  -- The greatest integer less than or equal to $3333\frac{1}{3}$ is 3333.
  rw [h₀]
  norm_num

-- Adding these values, we get $3 + 33 + 333 + 3333 = 3702$.
rw [h₁, h₂, h₃, h₄]
norm_num
