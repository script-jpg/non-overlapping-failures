-- We want to show that the set of 4-digit positive integers with only even digits and divisible by 5 has cardinality 100.

  -- The structure of the proof will be as follows:
  -- 1. Show that the set S has 100 elements by computing it directly using computational methods
  -- 2. This is feasible because the set is finite and not too large (100 elements is manageable)

  have h₁ : S.card = 100 := by
    -- Lean has a computational engine that can evaluate expressions involving finite sets and natural numbers
    -- We will use this to compute the cardinality of S directly
    have : S = Finset.filter (fun n => 1000 ≤ n ∧ n ≤ 9999 ∧ (∀ d : ℕ, d ∈ Nat.digits 10 n → Even d) ∧ 5 ∣ n) (Finset.Icc 1000 9999) := by

      simp [h₀]
      omega
    rw [this]

    -- Lean can compute the cardinality of this filtered set directly
    native_decide
  exact h₁
