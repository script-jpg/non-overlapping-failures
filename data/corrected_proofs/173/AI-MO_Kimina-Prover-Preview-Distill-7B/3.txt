-- We need to find the total number of marbles modulo 10.
calc (239 + 174 + 83) % 10
    -- The sum of the marbles is 239 + 174 + 83.
    _ = (413 + 83) % 10 := by simp
    -- 239 + 174 = 413.
    _ = 496 % 10 := by simp
    -- 413 + 83 = 496.
    _ = 6 := by norm_num
    -- The remainder when 496 is divided by 10 is 6.
