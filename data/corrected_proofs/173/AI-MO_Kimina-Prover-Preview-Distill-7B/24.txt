-- We need to find the number of marbles that need to be removed so that the remaining marbles can be divided into piles of 10.
-- First, we calculate the total number of marbles.
have h1 : 239 + 174 + 83 = 496 := by norm_num
-- Then, we find the remainder when 496 is divided by 10.
have h2 : 496 % 10 = 6 := by norm_num
-- Therefore, 6 marbles need to be removed.
rw [h1, h2]
