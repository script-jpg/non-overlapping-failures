-- We need to find the number of marbles that need to be removed so that the remaining marbles can be grouped into piles of 10.
-- First, let's calculate the total number of marbles.
let total_marbles := 239 + 174 + 83
-- Then, we need to find the remainder when the total number of marbles is divided by 10.
let marbles_to_remove := total_marbles % 10
-- Show that the number of marbles to remove is 6.
have h : marbles_to_remove = 6 := by
  norm_num [total_marbles, marbles_to_remove]
-- This is exactly what we need to prove.
exact h
