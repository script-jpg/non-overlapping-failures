-- Let's calculate $f(4)$.
have h1 : f 4 = -1 := by
  -- Since $n = 4$ is even, we use the second case of the function definition.
  specialize h₁ 4 (by decide)
  -- Plug $n = 4$ into the expression for even $n$.
  rw [h₁]
  -- Calculate $4^2 - 4 \cdot 4 - 1 = 16 - 16 - 1 = -1$.
  norm_num

-- Therefore, $f(4) = -1$.
exact h1
