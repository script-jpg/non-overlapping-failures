-- Given $f(x)=cx^3-9x+3$ and $f(2)=9$, find the value of $c$. Show that it is 3.

  -- f(2) = c * 2^3 - 9 * 2 + 3 = 8c - 18 + 3 = 8c - 15
  have : f 2 = c * 2 ^ 3 - 9 * 2 + 3 := h₀ 2
  rw [this] at h₁
  -- 8c - 15 = 9
  ring_nf at h₁
  -- 8c = 24
  have : c * 8 = 24 := by linarith
  -- c = 3
  linarith
