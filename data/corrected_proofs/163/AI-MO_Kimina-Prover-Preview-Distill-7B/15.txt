-- Given $f(x)=cx^3-9x+3$ and $f(2)=9$, find the value of $c$. Show that it is 3.

  -- Substitute x=2 in f(x) to get f(2)=c*2^3-9*2+3=8c-18+3=8c-15.
  have h2 : f 2 = c * 2 ^ 3 - 9 * 2 + 3 := h₀ 2

  -- Substitute h2 in h1 to get 8c-15=9.
  rw [h2] at h₁

  -- Solve the equation 8c-15=9 to get 8c=24.
  have h3 : c * 8 = 24 := by linarith

  -- Divide both sides by 8 to get c=3.
  have hc : c = 3 := by linarith

  -- Return the result.
  exact hc
