-- We remove the absolute value signs by considering the signs of the expressions inside them
have h₃ : x - p ≥ 0 := by linarith
have h₄ : x - 15 ≤ 0 := by linarith
have h₅ : x - p - 15 ≤ 0 := by linarith

-- Simplify the expressions and remove the absolute value signs
rw [abs_of_nonneg h₃, abs_of_nonpos h₄, abs_of_nonpos h₅] at h₂

-- Simplify the goal
linarith
