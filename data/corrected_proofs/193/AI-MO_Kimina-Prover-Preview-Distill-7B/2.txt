-- Show that $f(1) = \dfrac{1}{1 + 2} = \dfrac{1}{3}.$
have h₁ : f 1 = 1 / 3 := by
  specialize h₀ 1 (by norm_num)
  norm_num at h₀
  exact h₀

-- Show that $f(f(1)) = f\left(\dfrac{1}{3}\right) = \dfrac{1}{\frac{1}{3} + 2} = \dfrac{1}{\frac{7}{3}} = \dfrac{3}{7}.$
specialize h₀ (1 / 3) (by norm_num)
rw [h₁, h₀]
norm_num
