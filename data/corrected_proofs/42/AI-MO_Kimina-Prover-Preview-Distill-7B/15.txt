-- Calculate the sum of the numbers
let sum := 2000 + 2001 + 2002 + 2003 + 2004 + 2005 + 2006

-- Calculate the remainder when the sum is divided by 7
let remainder := sum % 7

-- Calculate the quotient when the sum is divided by 7
let quotient := sum / 7

-- Check that the remainder is 0
have h : remainder = 0 := by
  norm_num [sum, remainder, quotient]

-- Conclude that the remainder when $2000+2001+2002+2003+2004+2005+2006$ is divided by $7$ is 0
exact h
