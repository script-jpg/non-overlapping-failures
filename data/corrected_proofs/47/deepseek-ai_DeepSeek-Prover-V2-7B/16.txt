have h₀ : a * (2 - a) ≤ 1 := by
  -- Use the fact that the square of any real number is non-negative to prove the inequality.
  have h₁ : (a - 1) ^ 2 ≥ 0 := sq_nonneg _
  have h₂ : (a - 1) ^ 2 = a ^ 2 - 2 * a + 1 := by ring
  have h₃ : a ^ 2 - 2 * a + 1 ≥ 0 := by linarith
  have h₄ : a ^ 2 - 2 * a + 1 = a ^ 2 - 2 * a + 1 := by rfl
  nlinarith
exact h₀
